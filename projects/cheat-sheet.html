SQL INNER JOIN Keyword

SELECT Orders.OrderID, Customers.CustomerName, Orders.OrderDate
FROM Orders
INNER JOIN Customers
ON Orders.CustomerID=Customers.CustomerID;

Result

OrderID CustomerName OrderDate
---------- ---------- ----------
10308 Kelvin 1996-09-18
